# New Video Editing Tools Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 7 new FFmpeg-backed tools — `compress_video`, `generate_thumbnail`, `normalize_audio`, `fade_audio`, `add_watermark`, `create_gif`, `blur_region` — to ClipChat's engine and web UI.

**Architecture:** Each tool follows the established pattern: Zod schema in `types/tools.ts` → FFmpeg implementation in `ffmpeg/<file>.ts` → registered in `queue/worker.ts` (TOOL_MAP), `mcp/tools.ts` (MCP_TOOLS), and `api/routes/jobs.ts` (TOOL_SCHEMAS). The system prompt in `chat.ts` and the web `tool-call-card.tsx` are updated last.

**Tech Stack:** fluent-ffmpeg, Zod, Vitest, React (web), TypeScript throughout.

import ffmpeg from 'fluent-ffmpeg';
import { tempOutputPath, runFfmpegWithCleanup, ffprobePromise } from './executor.js';
import type { CompressVideoInput } from '../types/tools.js';

const PRESET_CONFIG: Record<string, {
  crf: number;
  ffmpegPreset: string;
  audioBitrate: string;
  maxHeight: number | null;
  codec: string;
}> = {
  web:      { crf: 23, ffmpegPreset: 'fast',    audioBitrate: '128k', maxHeight: null,  codec: 'libx264' },
  mobile:   { crf: 28, ffmpegPreset: 'faster',  audioBitrate: '96k',  maxHeight: 720,   codec: 'libx264' },
  whatsapp: { crf: 30, ffmpegPreset: 'fast',    audioBitrate: '128k', maxHeight: 720,   codec: 'libx264' },
  telegram: { crf: 26, ffmpegPreset: 'fast',    audioBitrate: '128k', maxHeight: null,  codec: 'libx264' },
  archive:  { crf: 28, ffmpegPreset: 'slow',    audioBitrate: '128k', maxHeight: null,  codec: 'libx265' },
};

export async function compressVideo(
  input: CompressVideoInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const output = tempOutputPath('mp4');
  const config = PRESET_CONFIG[input.preset];

  let cmd = ffmpeg(input.input_file)
    .videoCodec(config.codec)
    .outputOptions([`-crf ${config.crf}`, `-preset ${config.ffmpegPreset}`])
    .audioCodec('aac')
    .audioBitrate(config.audioBitrate);

  if (config.maxHeight) {
    // scale=-2:720 keeps aspect ratio and ensures width is divisible by 2
    cmd = cmd.videoFilter(`scale=-2:${config.maxHeight}`);
  }

  if (input.target_size_mb) {
    const probe = await ffprobePromise(input.input_file);
    const duration = Number(probe.format.duration ?? 1);
    const targetBitrate = Math.max(100, Math.floor((input.target_size_mb * 8192) / duration));
    cmd = cmd.outputOptions([
      `-b:v ${targetBitrate}k`,
      `-maxrate ${targetBitrate * 2}k`,
      `-bufsize ${targetBitrate * 4}k`,
    ]);
  }

  cmd = cmd.output(output);
  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}
```

import ffmpeg from 'fluent-ffmpeg';
import { tempOutputPath, runFfmpegWithCleanup } from './executor.js';
import type { GenerateThumbnailInput } from '../types/tools.js';

export async function generateThumbnail(
  input: GenerateThumbnailInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const format = input.format ?? 'jpg';
  const output = tempOutputPath(format);

  let cmd = ffmpeg(input.input_file)
    .seekInput(input.timestamp)
    .outputOptions(['-vframes 1']);

  if (input.width) {
    cmd = cmd.videoFilter(`scale=${input.width}:-1`);
  }

  if (format === 'jpg') cmd = cmd.outputOptions(['-q:v 2']);
  if (format === 'webp') cmd = cmd.outputOptions(['-quality 85']);

  cmd = cmd.output(output);
  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}
```

```typescript
import ffmpeg from 'fluent-ffmpeg';
import { tempOutputPath, runFfmpegWithCleanup, ffprobePromise } from './executor.js';
import { AppError } from '../types/job.js';
import type { NormalizeAudioInput, FadeAudioInput } from '../types/tools.js';

export async function normalizeAudio(
  input: NormalizeAudioInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const probe = await ffprobePromise(input.input_file);
  const hasAudio = probe.streams.some(s => s.codec_type === 'audio');
  if (!hasAudio) throw new AppError(400, 'no audio stream found in input file');

  const output = tempOutputPath('mp4');
  const lufs = input.target_lufs ?? -14;
  const tp = input.true_peak ?? -1;

  const cmd = ffmpeg(input.input_file)
    .audioFilter(`loudnorm=I=${lufs}:TP=${tp}:LRA=11`)
    .outputOptions(['-c:v copy'])
    .output(output);

  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}

export async function fadeAudio(
  input: FadeAudioInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const output = tempOutputPath('mp4');

  const probe = await ffprobePromise(input.input_file);
  const duration = Number(probe.format.duration ?? 0);
  const maxEach = duration / 2;

  const fadeIn = Math.min(input.fade_in_duration ?? 0, maxEach);
  const fadeOut = Math.min(input.fade_out_duration ?? 0, maxEach);

  if (fadeIn === 0 && fadeOut === 0) {
    // No-op: copy through
    const cmd = ffmpeg(input.input_file).outputOptions(['-c copy']).output(output);
    await runFfmpegWithCleanup(cmd, output, { onProgress });
    return output;
  }

  const filters: string[] = [];
  if (fadeIn > 0) filters.push(`afade=t=in:st=0:d=${fadeIn}`);
  if (fadeOut > 0) filters.push(`afade=t=out:st=${duration - fadeOut}:d=${fadeOut}`);

  const cmd = ffmpeg(input.input_file)
    .audioFilter(filters.join(','))
    .outputOptions(['-c:v copy'])
    .output(output);

  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}
```

import ffmpeg from 'fluent-ffmpeg';
import { tempOutputPath, runFfmpegWithCleanup } from './executor.js';
import type { AddWatermarkInput } from '../types/tools.js';

// Overlay position expressions using FFmpeg overlay filter vars:
// W/H = main video dims, w/h = overlay dims, M = margin value
const POSITION_EXPRS: Record<string, { x: string; y: string }> = {
  top_left:     { x: 'M',       y: 'M' },
  top_right:    { x: 'W-w-M',   y: 'M' },
  bottom_left:  { x: 'M',       y: 'H-h-M' },
  bottom_right: { x: 'W-w-M',   y: 'H-h-M' },
  center:       { x: '(W-w)/2', y: '(H-h)/2' },
};

export async function addWatermark(
  input: AddWatermarkInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const output = tempOutputPath('mp4');
  const pos = POSITION_EXPRS[input.position ?? 'bottom_right'];
  const scale = input.scale ?? 0.15;
  const opacity = input.opacity ?? 1;
  const margin = String(input.margin ?? 10);

  const x = pos.x.replace(/M/g, margin);
  const y = pos.y.replace(/M/g, margin);

  // Scale watermark, optionally apply opacity
  let wmFilter = `[1:v]scale=iw*${scale}:-1:flags=lanczos`;
  if (opacity < 1) {
    wmFilter += `,format=rgba,colorchannelmixer=aa=${opacity}`;
  }
  wmFilter += '[wm]';

  const cmd = ffmpeg(input.input_file)
    .input(input.watermark_file)
    .complexFilter([
      wmFilter,
      `[0:v][wm]overlay=${x}:${y}`,
    ])
    .outputOptions(['-c:a copy'])
    .output(output);

  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}

import ffmpeg from 'fluent-ffmpeg';
import fs from 'fs';
import { tempOutputPath, runFfmpeg, runFfmpegWithCleanup } from './executor.js';
import type { CreateGifInput } from '../types/tools.js';

export async function createGif(
  input: CreateGifInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const output = tempOutputPath('gif');
  const palette = tempOutputPath('png');
  const fps = input.fps ?? 10;
  const width = input.width ?? 480;
  const scaleFilter = `fps=${fps},scale=${width}:-1:flags=lanczos`;

  try {
    // Pass 1: generate optimised palette (reports 0–50% progress)
    const pass1 = ffmpeg(input.input_file);
    if (input.start_time) pass1.seekInput(input.start_time);
    if (input.end_time) pass1.outputOptions([`-to ${input.end_time}`]);
    pass1.videoFilter(`${scaleFilter},palettegen`).output(palette);
    await runFfmpeg(pass1, {
      onProgress: onProgress ? (p) => onProgress(p / 2) : undefined,
    });

    // Pass 2: encode GIF using palette (reports 50–100% progress)
    const ditheringOpts = input.optimize !== false
      ? 'dither=bayer:bayer_scale=5:diff_mode=rectangle'
      : 'dither=none';

    const pass2 = ffmpeg(input.input_file).input(palette);
    if (input.start_time) pass2.seekInput(input.start_time);
    if (input.end_time) pass2.outputOptions([`-to ${input.end_time}`]);
    pass2
      .complexFilter([`${scaleFilter}[x];[x][1:v]paletteuse=${ditheringOpts}`])
      .output(output);

    await runFfmpegWithCleanup(pass2, output, {
      onProgress: onProgress ? (p) => onProgress(50 + p / 2) : undefined,
    });
  } finally {
    await fs.promises.unlink(palette).catch(() => {});
  }

  return output;
}

import ffmpeg from 'fluent-ffmpeg';
import { tempOutputPath, runFfmpegWithCleanup, ffprobePromise } from './executor.js';
import type { BlurRegionInput } from '../types/tools.js';

export async function blurRegion(
  input: BlurRegionInput,
  onProgress?: (p: number) => void,
): Promise<string> {
  const output = tempOutputPath('mp4');

  const probe = await ffprobePromise(input.input_file);
  const stream = probe.streams.find(s => s.codec_type === 'video');
  const vw = stream?.width ?? 1920;
  const vh = stream?.height ?? 1080;

  // Start with preset or fall back to full frame
  let x = input.x ?? 0;
  let y = input.y ?? 0;
  let w = input.width ?? vw;
  let h = input.height ?? vh;

  if (input.preset && input.x === undefined && input.y === undefined) {
    if (input.preset === 'face_top_center') {
      w = Math.floor(vw * 0.5);
      h = Math.floor(vh * 0.4);
      x = Math.floor(vw * 0.25);
      y = 0;
    } else if (input.preset === 'lower_third') {
      w = vw;
      h = Math.floor(vh * 0.33);
      x = 0;
      y = Math.floor(vh * 0.67);
    } else {
      // full_frame
      w = vw; h = vh; x = 0; y = 0;
    }
  }

  const strength = input.blur_strength ?? 10;

  const enableClause = (input.start_time && input.end_time)
    ? `:enable='between(t,${input.start_time},${input.end_time})'`
    : '';

  const cmd = ffmpeg(input.input_file)
    .complexFilter([
      `[0:v]crop=${w}:${h}:${x}:${y},boxblur=${strength}:${strength}[blurred]`,
      `[0:v][blurred]overlay=${x}:${y}${enableClause}[out]`,
    ])
    .outputOptions(['-map [out]', '-map 0:a?', '-c:a copy'])
    .output(output);

  await runFfmpegWithCleanup(cmd, output, { onProgress });
  return output;
}

        src={src}
        controls
        className="h-full w-full object-contain bg-black"
        onError={() => setUseImageFallback(true)}
      />
      {labelEl}
    </div>
  );
}
